연습문제 정답

data는 diamonds

aes는 x = carat, y = price

geom_point 를 사용해서 scatter plot 그리기

library(ggplot2)

ggplot(data = diamonds, aes(x=carat, y=price))+
  geom_point()

data = iris

aes는 x = Species, y = Sepal.Width 일 때

geom_boxplot를 사용해서 boxplot 그리기

면의 색상(fill)은 #3399ff

선의 색상(colour 또는 color)은 #c8b923

ggplot(data = iris, aes(x = Species, y = Sepal.Width))+
  geom_boxplot(fill = '#3399ff', colour = '#c8b923')

data : diamonds

aes : x = clarity, fill = cut

geom : bar

position : stack

ggplot(data = diamonds, aes(x=clarity, fill=cut))+
  geom_bar(position='stack')

data : iris

aes : x = Sepal.Width, y = Sepal.Length

geom : point

facet : wrap(~Species)

ggplot(data = iris, aes(x=Sepal.Width, y=Sepal.Length))+
  geom_point()+
  facet_wrap(~Species)
