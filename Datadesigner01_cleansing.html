Pipe Operator 

pipe_result <- 
  streetlamp %>%
  melt(value.name = "count") %>%
  mutate(year = gsub("X|","",variable)) %>%
  select(location = X, year, count) %>%
  group_by(year) %>%
  mutate(mean = round(mean(count),2), 
         diff = round(count - mean, 2))
